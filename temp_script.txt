        function downloadLatencyData() {
            if (latencyData.length === 0) {
                alert("Ingen data å laste ned. Spill noen noter først!");
                return;
            }

            // Bygg CSV innhold - FORSTÅELIG VERSJON FOR RAPPORT
            let csvContent = "data:text/csv;charset=utf-8,";
            csvContent += "SANG,TID_I_SANG(SEK),HENDELSE,MIDI_NOTE,KRAFT(VEL),FORSINKELSE(MS),LOOKAHEAD_INNSTILLING(MS)\n";
            
            latencyData.forEach(function(row) {
                csvContent += `${row.song},${row.time_sec},${row.event},${row.note},${row.velocity},${row.latency_ms},${row.lookahead}\n`;
            });

            const encodedUri = encodeURI(csvContent);
            const link = document.createElement("a");
            link.setAttribute("href", encodedUri);
            link.setAttribute("download", `Klaviator_Rapport_${currentSongName}_${new Date().getTime()}.csv`);
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
            
            log('info', `Eksporterte ${latencyData.length} rader med forskningsdata.`);
        }

        function refreshLibrary() {
            fetch('/api/library')
                .then(res => res.json())
                .then(data => {
                    midiLibList.innerHTML = '';
                    if (!data.files || data.files.length === 0) {
                        midiLibList.innerHTML = '<div style="padding: 10px; color: #555;">Mappen er tom...</div>';
                        return;
                    }
                    data.files.forEach(file => {
                        const div = document.createElement('div');
                        div.style.padding = '8px 12px';
                        div.style.borderBottom = '1px solid #222';
                        div.style.cursor = 'pointer';
                        div.style.display = 'flex';
                        div.style.justifyContent = 'space-between';
                        div.style.fontSize = '0.85rem';
                        div.innerHTML = `<span>🎵 ${file}</span> <span style="color: var(--cyan)">SPILL ▶️</span>`;
                        div.onclick = () => playFromLibrary(file);
                        midiLibList.appendChild(div);
                    });
                })
                .catch(err => log('error', 'Kunne ikke laste bibliotek.'));
        }

        function playFromLibrary(filename) {
            log('info', `Starter bibliotek-avspilling: ${filename}`);
            currentSongName = filename;
            songStartTime = Date.now();
            fetch(`/api/play_library/${filename}`)
                .then(res => res.json())
                .then(data => {
                    document.getElementById('midi-status').innerText = filename;
                    document.getElementById('midi-status').style.color = "var(--gold-bright)";
                });
        }

        function connectWebSocket() {
            const serverIP = window.location.hostname;
            ws = new WebSocket(`ws://${serverIP}:8000/ws/unity`);

            ws.onopen = () => {
                statusIndicator.classList.add('connected');
                statusText.innerText = "TILKOBLET";
                statusText.style.color = "var(--success)";
                log('info', 'WebSocket-bro etablert med Hjernen.');
                refreshLibrary();
            };

            ws.onmessage = (event) => {
                const data = JSON.parse(event.data);
                
                if (data.event === "playback_start") {
                    currentSongName = data.data.file;
                    songStartTime = Date.now();
                    log('info', `START: ${data.data.file} (${data.data.total_notes} noter)`);
                    return;
                }

                if (data.event === "playback_finish") {
                    log('info', `FERDIG: Avspilling fullført.`);
                    return;
                }

                if (data.event === "bpm_change") {
                    document.getElementById('bpm-value').innerText = data.data.bpm;
                    log('info', `Tempo: ${data.data.bpm} BPM`);
                    return;
                }
                
                if (data.event === "system_halt") {
                    log('error', '!!! SYSTEM STOPPET - E-STOP AKTIVERT !!!');
                    if (pianoPlayer) pianoPlayer.stop();
                    return;
                }

                const latency = Date.now() - data.data.timestamp;
                recordLatency(latency, data.data.note, data.event, data.data.velocity);
                
                requestAnimationFrame(() => {
                    cpuLatencyText.innerText = latency + "ms";
                    if(latency > 25) { cpuLatencyText.style.color = "var(--danger)"; } 
                    else { cpuLatencyText.style.color = "var(--cyan)"; }
                });

                if(data.event === "note_on") {
                    handleNoteOn(data.data.note);
                    playNoteSound(data.data.note, data.data.velocity);
                    const btn = document.querySelector(`button[onclick="sendNote(${data.data.note})"]`);
                    if(btn) btn.classList.add('active');
                } else if(data.event === "note_off") {
                    handleNoteOff(data.data.note);
                    stopNoteSound(data.data.note);
                    const btn = document.querySelector(`button[onclick="sendNote(${data.data.note})"]`);
                    if(btn) btn.classList.remove('active');
                }
            };

            ws.onclose = () => {
                statusIndicator.classList.remove('connected');
                statusText.innerText = "PRØVER PÅ NYTT...";
                statusText.style.color = "var(--danger)";
                log('error', 'Forbindelse brutt. Prøver å koble til igjen...');
                setTimeout(connectWebSocket, 2000);
            };
        }

        function sendNote(note) {
            fetch(`/api/test_note/${note}`)
                .then(response => response.json())
                .catch(error => log('error', `FEIL: Node ${note} svarer ikke.`));
        }
        
        function uploadMidi() {
            const input = document.getElementById('midi-upload');
            if(input.files.length === 0) return;
            
            const file = input.files[0];
            currentSongName = file.name;
            songStartTime = Date.now();
            document.getElementById('midi-status').innerText = file.name;
            
            const formData = new FormData();
            formData.append('file', file);
            
            log('info', `Laster opp sekvens: ${file.name}...`);
            
            fetch('/api/upload_midi', { method: 'POST', body: formData })
            .then(res => res.json())
            .then(data => log('info', `Opplasting vellykket.`))
            .catch(err => log('error', 'Feil ved opplasting av MIDI.'));
        }

        function controlPlayback(action) {
            fetch(`/api/playback/${action}`)
                .then(res => res.json())
                .then(data => {
                    log('info', `Kommando: ${action.toUpperCase()}`);
                    if (action === 'stop' || action === 'pause') {
                        if (pianoPlayer) pianoPlayer.stop();
                    }
                })
                .catch(err => log('error', 'Kunne ikke kommunisere med Hjernen.'));
        }

        function triggerEStop() {
            log('error', '!!! EMERGENCY STOP !!!');
            fetch('/api/estop');
        }

        function log(type, message) {
            const time = new Date().toISOString().substring(11, 23);
            let colorClass = 'log-info';
            if (type === 'error') colorClass = 'log-error';
            if (type === 'send') colorClass = 'log-send';
            if (type === 'recv') colorClass = 'log-recv';

            const logHtml = `<span class="log-time">[${time}]</span> <span class="${colorClass}">${message}</span>`;
            const divMain = document.createElement('div');
            divMain.innerHTML = logHtml;
            logWindow.appendChild(divMain);
            
            if (logWindow.children.length > 50) logWindow.removeChild(logWindow.firstChild);
            logWindow.scrollTop = logWindow.scrollHeight;

            const divModal = document.createElement('div');
            divModal.innerHTML = logHtml;
            modalLogWindow.appendChild(divModal);
            if (modalLogWindow.children.length > 200) modalLogWindow.removeChild(modalLogWindow.firstChild);
            modalLogWindow.scrollTop = modalLogWindow.scrollHeight;
        }
